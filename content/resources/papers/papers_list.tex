% Load IEEE style and bibliography file
\usepackage[style=ieee,citestyle=numeric-comp]{biblatex}
\addbibresource{papers.bib}

% Define custom command to print a single paper's bibliography entry
\newcommand{\printpaperbib}[1]{%
  \begin{refsection} % Start a new reference section
    \nocite{#1}  % Add only the specified paper to the bibliography
    \printbibliography[heading=none]  % Print only the cited paper
  \end{refsection}
}

\begin{document}

\author{}
\date{}

\maketitle

\section*{2023}

% Print the bibliography for a single paper
\printpaperbib{example2023}

% Print the bibliography for another specific paper
